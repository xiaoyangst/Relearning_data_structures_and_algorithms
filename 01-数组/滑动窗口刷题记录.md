## 209.长度最小的子数组

```c++
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int result = INT32_MAX;
        int sum = 0;
        int fast_Index = 0;
        int slow_Index = 0;

        for (int num:nums) sum += num;

        if (sum < target) return 0;

        sum = 0;

        for (fast_Index = 0; fast_Index < nums.size(); fast_Index++) {
            sum += nums[fast_Index];
            while (sum >= target){  //滑动窗口
                result = (fast_Index - slow_Index + 1) < result ? (fast_Index - slow_Index + 1) : result;
                sum -= nums[slow_Index];
                slow_Index++;
            }
        }
        result = (fast_Index - slow_Index + 1) < result ? (fast_Index - slow_Index + 1) : result;
        return result;
    }
};
```

非固定滑动窗口，通过不断缩小窗口找到符合条件

## 239.滑动窗口最大值

```c++
class Solution {
public:
    int get_Max_Index(vector<int>& nums,int left,int right){
        int result = INT32_MIN;
        int index = 0;
        for (int i = left; i <= right; i++) {
            if (nums[i] > result){
                result = nums[i];
                index = i;
            }
        }
        return index;
    }
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int slow_Index = 0;
        int fast_Index = 0;
        vector<int> result;
        int tmp = INT32_MIN;
        int index = 0;

        if (nums.size() < k || k == 1) return nums;

        for (fast_Index = k - 1; fast_Index < nums.size();) {
            index = get_Max_Index(nums,slow_Index,fast_Index);
            tmp = nums[index];
            while (fast_Index < nums.size() && tmp >= nums[fast_Index] && index >= slow_Index && index <= fast_Index){
                    result.push_back(tmp);
                    fast_Index++;
                    slow_Index++;
            }
        }

        return result;
    }
};
```

从题目描述来看，就知道这个是固定窗口

这道题**会超时**，不能用滑动窗口来做，应该用单调队列

## LCR 016.不含重复字符的最长子字符串

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        // 哈希集合，记录每个字符是否出现过
        unordered_set<char> occ;
        int n = s.size();
        // 右指针，初始值为 -1，相当于我们在字符串的左边界的左侧，还没有开始移动
        int rk = -1, ans = 0;
        // 枚举左指针的位置，初始值隐性地表示为 -1
        for (int i = 0; i < n; ++i) {
            if (i != 0) {
                // 左指针向右移动一格，移除一个字符
                occ.erase(s[i - 1]);
            }
            while (rk + 1 < n && !occ.count(s[rk + 1])) {
                // 不断地移动右指针
                occ.insert(s[rk + 1]);
                ++rk;
            }
            // 第 i 到 rk 个字符是一个极长的无重复字符子串
            ans = max(ans, rk - i + 1);
        }
        return ans;
    }
};
```
